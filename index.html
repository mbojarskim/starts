const express = require('express'); const fs = require('fs'); const path = require('path'); const app = express(); const port = 3000; // CORS - pozwala telefonom na wysyłanie app.use((req, res, next) => { res.header('Access-Control-Allow-Origin', '*'); res.header('Access-Control-Allow-Headers', 'Origin, X-Requested-With, Content-Type, Accept'); res.header('Access-Control-Allow-Methods', 'GET, POST, DELETE, OPTIONS'); if (req.method === 'OPTIONS') { return res.sendStatus(200); } next(); }); app.use(express.json()); // Udostępnianie plików statycznych (w tym start.mp3) app.use(express.static(__dirname)); let starts = []; // Zapis startu (czas z telefonu - moment naciśnięcia) app.post('/api/start', (req, res) => { const { raceNumber, time, timestamp, latencyCompensation } = req.body; const startRecord = { id: Date.now(), raceNumber: raceNumber, time: time, // czas z momentu naciśnięcia! serverTime: new Date().toISOString(), timestamp: timestamp, latencyCompensation: latencyCompensation || 0 }; starts.push(startRecord); // Backup do pliku fs.writeFileSync('starts.json', JSON.stringify(starts, null, 2)); console.log(`\n🏁 ${new Date().toLocaleTimeString()} - ODEBRANO START!`); console.log(` Bieg: ${raceNumber}`); console.log(` Czas startu: ${time}`); console.log(` Kompensacja: ${latencyCompensation || 0}ms`); res.json({ success: true, id: startRecord.id }); }); // Odczyt wszystkich startów app.get('/api/starts', (req, res) => { res.json(starts); }); // Kasowanie wszystkich startów app.delete('/api/starts', (req, res) => { starts = []; fs.writeFileSync('starts.json', JSON.stringify(starts)); console.log('\n🗑️ Wszystkie starty zostały usunięte!'); res.json({ success: true }); }); // Informacja o stanie app.get('/api/status', (req, res) => { res.json({ status: 'running', startCount: starts.length, serverTime: new Date().toISOString() }); }); // Uruchomienie serwera app.listen(port, '0.0.0.0', () => { console.log(`\n${'='.repeat(50)}`); console.log(`✅ SYSTEM POMIARU CZASU - SERWER DZIAŁA`); console.log(`${'='.repeat(50)}`); console.log(`📱 Adres dla telefonu: http://TWOJ_IP:${port}`); console.log(`🔊 Sprawdź dźwięk: http://TWOJ_IP:${port}/start.mp3`); console.log(`📊 Wyniki: http://localhost:${port}/api/starts`); console.log(`${'='.repeat(50)}\n`); });
